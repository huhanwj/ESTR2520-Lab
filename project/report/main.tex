\begin{algorithm}[H]
  \SetAlgoLined
  \KwData{Set parameters: M, B, $R_{d}$, $\boldsymbol{w}$; modifiable parameters: a, $\delta,~\upsilon,~ \lambda$, step size $\gamma$}
  \KwResult{Output vector: $\boldsymbol{x}$}
  initialization\;
  $X = \boldsymbol{0}$\;
  \For{$i\leftarrow 0$ \KwTo $number~of~epochs$}{
  $\boldsymbol{x}^{(k+1)}=Proj_{X}{x^{(k)}-\gamma \cdot \nabla F(\boldsymbol{x}^{(k)})}$\;
  
  $objective~value=F(\boldsymbol{x}^{(k+1)})$\;
  
  $store~objective~value$
  }  
  output $x=Find \_ Best \_ X(list)$\;
\caption{The iterative algorithm by PGD}
\end{algorithm}
